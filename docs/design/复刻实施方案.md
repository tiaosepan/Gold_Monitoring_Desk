### 4.2 新浪财经API封装

```python
# utils/sina_api.py
import aiohttp
import re
from typing import Dict, Optional

class SinaFinanceAPI:
    """新浪财经API封装"""
    
    BASE_URL = "https://hq.sinajs.cn/list="
    
    @staticmethod
    async def fetch_gold_data() -> Dict:
        """
        获取黄金相关数据
        
        Returns:
            {
                'shfe_price': float,      # 沪金99价格（元/克）
                'london_price': float,    # 伦敦金价格（美元/盎司）
                'usdcny_rate': float      # 美元人民币汇率
            }
        """
        codes = [
            'gds_AU9999',  # 沪金99
            'hf_XAU',      # 伦敦金现货
            'USDCNY'       # 美元人民币
        ]
        
        url = f"{SinaFinanceAPI.BASE_URL}{','.join(codes)}"
        
        async with aiohttp.ClientSession() as session:
            async with session.get(url, timeout=aiohttp.ClientTimeout(total=10)) as response:
                text = await response.text()
                return SinaFinanceAPI.parse_response(text)
    
    @staticmethod
    def parse_response(text: str) -> Dict:
        """解析新浪API响应"""
        data = {}
        
        # 解析沪金99
        match = re.search(r'var hq_str_gds_AU9999="([^"]+)"', text)
        if match:
            fields = match.group(1).split(',')
            data['shfe_price'] = float(fields[0])
        
        # 解析伦敦金
        match = re.search(r'var hq_str_hf_XAU="([^"]+)"', text)
        if match:
            fields = match.group(1).split(',')
            data['london_price'] = float(fields[0])
        
        # 解析汇率
        match = re.search(r'var hq_str_USDCNY="([^"]+)"', text)
        if match:
            fields = match.group(1).split(',')
            data['usdcny_rate'] = float(fields[1])
        
        data['raw_payload'] = text
        data['note'] = '国内源: 新浪 gds_AU9999 沪金99；国际源: 新浪 hf_XAU 伦敦金现货'
        
        return data
```

### 4.3 溢价计算逻辑

```python
# services/sge_monitor.py
def calculate_premium(shfe_price: float, london_price: float, usdcny_rate: float) -> Dict:
    """
    计算SGE溢价
    
    Args:
        shfe_price: 上海金价（元/克）
        london_price: 伦敦金价（美元/盎司）
        usdcny_rate: 美元兑人民币汇率
    
    Returns:
        {
            'london_price_cny_per_g': float,  # 国际金折算（元/克）
            'premium_cny_per_g': float        # 溢价（元/克）
        }
    """
    # 1盎司 = 31.1035克
    london_price_cny_per_g = (london_price * usdcny_rate) / 31.1035
    premium_cny_per_g = shfe_price - london_price_cny_per_g
    
    return {
        'london_price_cny_per_g': round(london_price_cny_per_g, 4),
        'premium_cny_per_g': round(premium_cny_per_g, 4)
    }
```

### 4.4 市场开盘判断

```python
# utils/helpers.py
from datetime import datetime
import pytz

def check_market_status() -> Dict:
    """
    判断市场开盘状态
    
    Returns:
        {
            'shfe_market_open': bool,
            'london_market_open': bool,
            'both_markets_open': bool
        }
    """
    now = datetime.now(pytz.UTC)
    
    # 上海市场时间（北京时间）
    beijing_tz = pytz.timezone('Asia/Shanghai')
    beijing_time = now.astimezone(beijing_tz)
    hour = beijing_time.hour
    minute = beijing_time.minute
    weekday = beijing_time.weekday()
    
    # 上海金交所交易时间（周一到周五）
    # 上午：09:00-11:30
    # 下午：13:30-15:30
    shfe_open = False
    if weekday < 5:  # 周一到周五
        if (9 <= hour < 11) or (hour == 11 and minute <= 30):
            shfe_open = True
        elif (13 <= hour < 15) or (hour == 15 and minute <= 30):
            shfe_open = True
    
    # 伦敦金市场（24小时，周末除外）
    london_open = weekday < 5
    
    both_open = shfe_open and london_open
    
    return {
        'shfe_market_open': shfe_open,
        'london_market_open': london_open,
        'both_markets_open': both_open
    }
```

### 4.5 钉钉推送实现

```python
# utils/dingtalk.py
import time
import hmac
import hashlib
import base64
import urllib.parse
import aiohttp
import json

class DingTalkPusher:
    """钉钉机器人推送"""
    
    @staticmethod
    def generate_sign(secret: str, timestamp: int) -> str:
        """
        生成钉钉签名
        
        Args:
            secret: 密钥
            timestamp: 时间戳（毫秒）
        
        Returns:
            签名字符串
        """
        secret_enc = secret.encode('utf-8')
        string_to_sign = f'{timestamp}\n{secret}'
        string_to_sign_enc = string_to_sign.encode('utf-8')
        
        hmac_code = hmac.new(
            secret_enc,
            string_to_sign_enc,
            digestmod=hashlib.sha256
        ).digest()
        
        sign = urllib.parse.quote_plus(base64.b64encode(hmac_code))
        return sign
    
    @staticmethod
    async def send_message(webhook: str, secret: str, message: Dict) -> Dict:
        """
        发送钉钉消息
        
        Args:
            webhook: Webhook URL
            secret: 密钥
            message: 消息内容
        
        Returns:
            {
                'success': bool,
                'error': str (optional)
            }
        """
        timestamp = int(time.time() * 1000)
        sign = DingTalkPusher.generate_sign(secret, timestamp)
        
        url = f"{webhook}&timestamp={timestamp}&sign={sign}"
        
        try:
            async with aiohttp.ClientSession() as session:
                async with session.post(
                    url,
                    json=message,
                    headers={'Content-Type': 'application/json'},
                    timeout=aiohttp.ClientTimeout(total=10)
                ) as response:
                    result = await response.json()
                    
                    if result.get('errcode') == 0:
                        return {'success': True}
                    else:
                        return {
                            'success': False,
                            'error': result.get('errmsg', 'Unknown error')
                        }
        except Exception as e:
            return {
                'success': False,
                'error': str(e)
            }
```

### 4.6 反转等级判断

```python
# services/reversal_detection.py
def determine_reversal_level(conditions: Dict) -> int:
    """
    根据四类条件确定反转等级
    
    Args:
        conditions: {
            'price': bool,
            'political': bool,
            'war': bool,
            'us10y': bool
        }
    
    Returns:
        level: 0-4（1最强，4最弱，0无信号）
    """
    hit_count = sum(conditions.values())
    
    level_map = {
        4: 1,  # 四个全中 → 一级反转
        3: 2,  # 三个命中 → 二级反转
        2: 3,  # 两个命中 → 三级反转
        1: 4,  # 一个命中 → 四级反转
        0: 0   # 无命中 → 无信号
    }
    
    return level_map[hit_count]

def should_push_alert(level: int) -> bool:
    """
    判断是否应该推送
    
    Args:
        level: 反转等级
    
    Returns:
        是否推送（只推送1级和2级）
    """
    return level in [1, 2]
```

### 9.1 价格反转检测（Price条件）

```python
async def detect_price_condition(config: Dict) -> bool:
    """
    检测价格反转条件
    
    Args:
        config: {
            'lookback_window': int,      # 回看窗口（分钟）
            'min_bounce_pct': float,     # 最小反弹幅度（%）
            'ma_window': int             # 均线窗口（样本数）
        }
    
    Returns:
        是否检测到价格反转
    """
    # 1. 获取最近N分钟的金价数据
    lookback_minutes = config['lookback_window']
    cutoff_time = datetime.now() - timedelta(minutes=lookback_minutes)
    
    samples = db.query(SGESample).filter(
        SGESample.fetched_at >= cutoff_time
    ).order_by(SGESample.fetched_at).all()
    
    if len(samples) < config['ma_window']:
        return False
    
    # 2. 提取价格序列（使用伦敦金价格）
    prices = [s.london_price_usd_per_oz for s in samples]
    
    # 3. 计算移动平均线
    ma_window = config['ma_window']
    ma_values = []
    for i in range(ma_window - 1, len(prices)):
        ma = sum(prices[i - ma_window + 1:i + 1]) / ma_window
        ma_values.append(ma)
    
    if not ma_values:
        return False
    
    # 4. 找到窗口内的最低点
    min_price = min(prices)
    current_price = prices[-1]
    
    # 5. 计算反弹幅度
    bounce_pct = ((current_price - min_price) / min_price) * 100
    
    # 6. 判断反转条件
    # - 价格从低点反弹超过阈值
    # - 当前价格高于最近的移动平均线
    is_reversal = (
        bounce_pct >= config['min_bounce_pct'] and
        current_price > ma_values[-1]
    )
    
    return is_reversal
```

```python
            'score': int (1-10),
            'keywords': list
        }
    """
    text = (title + ' ' + summary).lower()
    
    # 政治缓和关键词
    political_keywords = [
        '和平', '谈判', '协议', '缓和', '外交', '对话', '停火', '休战',
        'peace', 'negotiation', 'agreement', 'diplomacy', 'ceasefire', 'truce'
    ]
    
    # 战争升级关键词
    war_keywords = [
        '战争', '冲突', '军事', '袭击', '空袭', '导弹', '入侵', '进攻', '轰炸',
        'war', 'conflict', 'military', 'attack', 'strike', 'missile', 'invasion'
    ]
    
    # 检测命中的关键词
    political_hits = [kw for kw in political_keywords if kw in text]
    war_hits = [kw for kw in war_keywords if kw in text]
    
    # 分类和评分
    if political_hits:
        # 政治缓和：分数越低，黄金越承压
        # 但根据更新日志："地缘缓和高分，紧张升级低分"
        # 所以政治缓和应该是高分（7-10分）
        intensity = len(political_hits)
        score = min(10, 7 + intensity)
        return {
            'category': 'political',
            'score': score,
            'keywords': political_hits
        }
    elif war_hits:
        # 战争升级：分数越低，黄金越承压
        # 所以战争升级应该是低分（1-4分）
        intensity = len(war_hits)
        score = max(1, 4 - intensity)
        return {
            'category': 'war',
            'score': score,
            'keywords': war_hits
        }
    else:
        return {
            'category': None,
            'score': 5,
            'keywords': []
        }
```

**注意**：评分规则根据更新日志修正：
- "停火+降息" = 10分（地缘缓和，利好黄金）
- "战争升级+加息" = 1分（紧张升级，黄金承压）

### 9.3 美债回落检测（US10Y条件）

```python
# services/us10y_monitor.py
async def detect_us10y_drop(config: Dict) -> bool:
    """
    检测美债收益率回落
    
    Args:
        config: {
            'lookback_hours': float,      # 回落窗口（小时）
            'drop_threshold_bp': float    # 回落阈值（基点）
        }
    
    Returns:
        是否检测到回落
    """
    # 1. 获取回落窗口内的数据
    lookback_hours = config['lookback_hours']
    cutoff_time = datetime.now() - timedelta(hours=lookback_hours)
    
    samples = db.query(US10YSample).filter(
        US10YSample.timestamp >= cutoff_time,
        US10YSample.tenor == '10y'
    ).order_by(US10YSample.timestamp).all()
    
    if len(samples) < 2:
        return False
    
    # 2. 提取收益率序列
    yields = [s.yield_rate for s in samples]
    
    # 3. 找到窗口内的最高点
    max_yield = max(yields)
    current_yield = yields[-1]
    
    # 4. 计算回落幅度（基点）
    drop_bp = (max_yield - current_yield) * 100
    
    # 5. 判断是否超过阈值
    is_drop = drop_bp >= config['drop_threshold_bp']
    
    return is_drop
```

```javascript
// 绑定SGE图表Y轴缩放
function bindSgeYAxisZoom(chart) {
    if (!chart || chart.__sgeYAxisWheelBound) return;
    chart.__sgeYAxisWheelBound = true;
    
    chart.getZr().on('mousewheel', (params) => {
        const nativeEvent = params?.event?.event;
        const wheelDelta = nativeEvent?.wheelDelta || -(nativeEvent?.detail || 0);
        
        // 获取鼠标在Y轴上的相对位置
        const chartHeight = chart.getHeight();
        const mouseY = params.offsetY;
        const anchorRatio = 1 - (mouseY / chartHeight);
        
        // 缩放Y轴
        zoomSgeYAxisByWheel(wheelDelta, anchorRatio);
        
        // 阻止默认滚动
        nativeEvent?.preventDefault();
    });
}

function zoomSgeYAxisByWheel(delta, anchorRatio = 0.5) {
    const current = state.sgeYAxisRange || { min: 800, max: 1200 };
    const span = Math.max(20, current.max - current.min);
    
    // 缩放因子
    const factor = delta > 0 ? 0.9 : 1.1;
    const newSpan = Math.min(4000, Math.max(20, span * factor));
    
    // 计算新的范围（保持锚点位置不变）
    const ratio = Math.min(1, Math.max(0, anchorRatio));
    const center = current.min + span * ratio;
    const nextMin = center - newSpan * ratio;
    const nextMax = center + newSpan * (1 - ratio);
    
    state.sgeYAxisRange = {
        min: Number(nextMin.toFixed(2)),
        max: Number(nextMax.toFixed(2))
    };
    
    // 重新渲染图表
    renderSgeChartsOnly();
}
```
