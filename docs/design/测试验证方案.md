```python
import feedparser
import aiohttp

async def fetch_rss_feed(url: str) -> list:
    """
    获取RSS源数据
    
    Args:
        url: RSS源地址
    
    Returns:
        事件列表
    """
    async with aiohttp.ClientSession() as session:
        async with session.get(url, timeout=10) as response:
            content = await response.text()
            
    # 解析RSS
    feed = feedparser.parse(content)
    
    events = []
    for entry in feed.entries:
        events.append({
            'title': entry.get('title', ''),
            'summary': entry.get('summary', ''),
            'link': entry.get('link', ''),
            'published': entry.get('published', ''),
            'published_parsed': entry.get('published_parsed', None)
        })
    
    return events
```

## 四、自动化对比测试脚本

### 4.1 API对比测试脚本

```python
# test_api_comparison.py
import requests
import json
from datetime import datetime

ORIGINAL_BASE = "http://rev.sccit.com.cn:8000"
REPLICA_BASE = "http://localhost:8000"

class APIComparator:
    """API对比测试工具"""
    
    def __init__(self):
        self.results = []
    
    def compare_endpoint(self, endpoint: str, params: dict = None):
        """对比单个API端点"""
        print(f"\n测试: {endpoint}")
        
        try:
            # 请求原系统
            original_resp = requests.get(
                f"{ORIGINAL_BASE}{endpoint}",
                params=params,
                timeout=10
            )
            
            # 请求复刻系统
            replica_resp = requests.get(
                f"{REPLICA_BASE}{endpoint}",
                params=params,
                timeout=10
            )
            
            # 对比状态码
            if original_resp.status_code != replica_resp.status_code:
                print(f"❌ 状态码不一致: {original_resp.status_code} vs {replica_resp.status_code}")
                self.results.append({
                    'endpoint': endpoint,
                    'status': 'FAIL',
                    'reason': 'Status code mismatch'
                })
                return False
            
            # 对比响应结构
            original_data = original_resp.json()
            replica_data = replica_resp.json()
            
            if self.compare_structure(original_data, replica_data):
                print(f"✅ 结构一致")
                self.results.append({
                    'endpoint': endpoint,
                    'status': 'PASS'
                })
                return True
            else:
                print(f"❌ 结构不一致")
                self.results.append({
                    'endpoint': endpoint,
                    'status': 'FAIL',
                    'reason': 'Structure mismatch'
                })
                return False
                
        except Exception as e:
            print(f"❌ 测试失败: {e}")
            self.results.append({
                'endpoint': endpoint,
                'status': 'ERROR',
                'reason': str(e)
            })
            return False
    
    def compare_structure(self, obj1, obj2, path="root") -> bool:
        """递归对比JSON结构"""
        if type(obj1) != type(obj2):
            print(f"  类型不一致 {path}: {type(obj1).__name__} vs {type(obj2).__name__}")
            return False
        
        if isinstance(obj1, dict):
            keys1 = set(obj1.keys())
            keys2 = set(obj2.keys())
            
            if keys1 != keys2:
                missing_in_replica = keys1 - keys2
                extra_in_replica = keys2 - keys1
                if missing_in_replica:
                    print(f"  复刻系统缺少字段 {path}: {missing_in_replica}")
                if extra_in_replica:
                    print(f"  复刻系统多余字段 {path}: {extra_in_replica}")
                return False
            
            for key in keys1:
                if not self.compare_structure(obj1[key], obj2[key], f"{path}.{key}"):
                    return False
        
        elif isinstance(obj1, list):
            if len(obj1) > 0 and len(obj2) > 0:
                return self.compare_structure(obj1[0], obj2[0], f"{path}[0]")
        
        return True
    
    def run_all_tests(self):
        """运行所有API对比测试"""
        print("=" * 60)
        print("API对比测试开始")
        print("=" * 60)
        
        # 测试所有端点
        endpoints = [
            ("/api/status", None),
            ("/api/history", {"range": "1H"}),
            ("/api/history", {"range": "1D"}),
            ("/api/history", {"range": "1W"}),
            ("/api/reversal/status", None),
            ("/api/reversal/history", {"range": "1D"}),
            ("/api/us10y/status", None),
            ("/api/us10y/history", {"range": "1D"}),
            ("/api/notification/logs", {"limit": 10}),
        ]
        
        for endpoint, params in endpoints:
            self.compare_endpoint(endpoint, params)
        
        # 输出测试报告
        print("\n" + "=" * 60)
        print("测试报告")
        print("=" * 60)
        
        total = len(self.results)
        passed = sum(1 for r in self.results if r['status'] == 'PASS')
        failed = sum(1 for r in self.results if r['status'] == 'FAIL')
        errors = sum(1 for r in self.results if r['status'] == 'ERROR')
        
        print(f"总计: {total}")
        print(f"✅ 通过: {passed}")
        print(f"❌ 失败: {failed}")
        print(f"⚠️  错误: {errors}")
        print(f"通过率: {passed/total*100:.1f}%")
        
        return passed == total

if __name__ == "__main__":
    comparator = APIComparator()
    success = comparator.run_all_tests()
    exit(0 if success else 1)
```

```python
import requests

def test_premium_calculation():
    """验证溢价计算准确性"""
    print("\n测试: 溢价计算")
    
    # 获取原系统数据
    original = requests.get('http://rev.sccit.com.cn:8000/api/status').json()
    sample = original['latest_sample']
    
    # 提取数据
    shfe_price = sample['shfe_price_cny_per_g']
    london_price = sample['london_price_usd_per_oz']
    usdcny = sample['usdcny_rate']
    
    # 手动计算溢价
    london_cny = (london_price * usdcny) / 31.1035
    premium = shfe_price - london_cny
    
    # 对比原系统的计算结果
    original_premium = sample['premium_cny_per_g']
    diff = abs(premium - original_premium)
    
    print(f"  沪金99: {shfe_price} 元/克")
    print(f"  伦敦金: {london_price} 美元/盎司")
    print(f"  汇率: {usdcny}")
    print(f"  国际金折算: {london_cny:.4f} 元/克")
    print(f"  手动计算溢价: {premium:.4f} 元/克")
    print(f"  原系统溢价: {original_premium} 元/克")
    print(f"  差异: {diff:.6f} 元/克")
    
    if diff < 0.01:
        print("✅ 溢价计算准确")
        return True
    else:
        print("❌ 溢价计算有误差")
        return False

def test_reversal_level():
    """验证反转等级判断"""
    print("\n测试: 反转等级判断")
    
    # 测试用例
    test_cases = [
        ({'price': True, 'political': True, 'war': True, 'us10y': True}, 1),
        ({'price': True, 'political': True, 'war': True, 'us10y': False}, 2),
        ({'price': True, 'political': True, 'war': False, 'us10y': False}, 3),
        ({'price': True, 'political': False, 'war': False, 'us10y': False}, 4),
        ({'price': False, 'political': False, 'war': False, 'us10y': False}, 0),
    ]
    
    def determine_level(conditions):
        hit_count = sum(conditions.values())
        level_map = {4: 1, 3: 2, 2: 3, 1: 4, 0: 0}
        return level_map[hit_count]
    
    all_passed = True
    for conditions, expected_level in test_cases:
        level = determine_level(conditions)
        status = "✅" if level == expected_level else "❌"
        print(f"  {status} 条件={conditions} → 等级={level} (期望={expected_level})")
        if level != expected_level:
            all_passed = False
    
    return all_passed

if __name__ == "__main__":
    test_premium_calculation()
    test_reversal_level()
```

```python
import time
import hmac
import hashlib
import base64
import urllib.parse
import requests
import json

def generate_dingtalk_sign(secret: str, timestamp: int) -> str:
    """生成钉钉签名"""
    secret_enc = secret.encode('utf-8')
    string_to_sign = f'{timestamp}\n{secret}'
    string_to_sign_enc = string_to_sign.encode('utf-8')
    
    hmac_code = hmac.new(
        secret_enc,
        string_to_sign_enc,
        digestmod=hashlib.sha256
    ).digest()
    
    sign = urllib.parse.quote_plus(base64.b64encode(hmac_code))
    return sign

def test_dingtalk_push(webhook: str, secret: str):
```

```python
async def concurrent_requests(url: str, count: int = 50):
    """并发请求测试"""
```

```python
        """
        采集所有RSS源
        
        Args:
            sources: RSS源URL列表
        
        Returns:
            {
                'events': [...],
                'error_count': int,
                'duration_ms': int
            }
        """
        start_time = datetime.now()
        events = []
        error_count = 0
        
        for url in sources:
            try:
                # 解析RSS
                feed_events = await self.parse_feed(url)
                
                # 处理每个事件
                for event in feed_events:
                    # 分类
                    category, keywords = self.classify_event(
                        event['title'],
                        event['summary']
                    )
                    
                    # 只保存命中反转上下文的事件
                    if category in ['political', 'war']:
                        # 评分
                        score = self.score_event(category, keywords, event['title'] + event['summary'])
                        
                        events.append({
                            'source_url': url,
                            'title': event['title'],
                            'summary': event['summary'],
                            'link': event['link'],
                            'category': category,
                            'score': score,
                            'keywords': ','.join(keywords),
                            'published_at': event['published']
                        })
                
            except Exception as e:
                error_count += 1
                print(f"RSS源采集失败 {url}: {e}")
        
        duration_ms = int((datetime.now() - start_time).total_seconds() * 1000)
        
        return {
            'events': events,
            'error_count': error_count,
            'duration_ms': duration_ms
        }
    
    async def parse_feed(self, url: str) -> List[Dict]:
        """解析RSS源"""
        async with aiohttp.ClientSession() as session:
            async with session.get(url, timeout=10) as response:
                content = await response.text()
        
        feed = feedparser.parse(content)
        
        events = []
        for entry in feed.entries:
            events.append({
                'title': entry.get('title', ''),
                'summary': entry.get('summary', ''),
                'link': entry.get('link', ''),
                'published': entry.get('published', '')
            })
        
        return events
    
    def classify_event(self, title: str, summary: str) -> tuple:
        """事件分类"""
        text = (title + ' ' + summary).lower()
        
        political_hits = [kw for kw in POLITICAL_KEYWORDS if kw.lower() in text]
        war_hits = [kw for kw in WAR_KEYWORDS if kw.lower() in text]
        
        if political_hits:
            return 'political', political_hits
        elif war_hits:
            return 'war', war_hits
        else:
            return None, []
    
    def score_event(self, category: str, keywords: list, text: str) -> int:
        """
        事件评分（1-10分）
        
        根据更新日志：
        - "停火+降息" = 10分（地缘缓和，利好黄金）
        - "战争升级+加息" = 1分（紧张升级，黄金承压）
        """
        if category == 'political':
            # 政治缓和：7-10分
            base_score = 7
            bonus = min(3, len(keywords))
            return base_score + bonus
        elif category == 'war':
            # 战争升级：1-4分
            base_score = 4
            penalty = min(3, len(keywords))
            return base_score - penalty
        else:
            return 5
```

## 十七、常见问题和解决方案

### 17.1 数据不一致问题

**问题**：复刻系统的溢价与原系统不一致

**排查步骤**：
1. 检查新浪API返回的原始数据
2. 检查解析逻辑（字段索引是否正确）
3. 检查计算公式（31.1035常量）
4. 检查浮点数精度

**解决方案**：
```python
# 使用Decimal确保精度
from decimal import Decimal, ROUND_HALF_UP

def calculate_premium_precise(shfe_price, london_price, usdcny):
    """高精度溢价计算"""
    shfe = Decimal(str(shfe_price))
    london = Decimal(str(london_price))
    rate = Decimal(str(usdcny))
    oz_to_g = Decimal('31.1035')
    
    london_cny = (london * rate / oz_to_g).quantize(
        Decimal('0.0001'),
        rounding=ROUND_HALF_UP
    )
    
    premium = (shfe - london_cny).quantize(
        Decimal('0.0001'),
        rounding=ROUND_HALF_UP
    )
    
    return float(premium)
```

```python
import requests
import json

def take_snapshot(base_url: str, name: str):
    """获取系统数据快照"""
    snapshot = {
        'timestamp': datetime.now().isoformat(),
        'system': name,
        'data': {}
    }

        '/api/status',

        '/api/reversal/status',
        '/api/reversal/history?range=1D',
        '/api/us10y/status',
        '/api/us10y/history?range=1D'
    ]

            response = requests.get(f"{base_url}{endpoint}", timeout=10)
            snapshot['data'][endpoint] = response.json()
        except Exception as e:
            snapshot['data'][endpoint] = {'error': str(e)}
    
    # 保存快照

        json.dump(snapshot, f, ensure_ascii=False, indent=2)
    
    print(f"快照已保存: {filename}")
    return snapshot
```
