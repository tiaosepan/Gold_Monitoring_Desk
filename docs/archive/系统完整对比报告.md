### B. 反转历史API格式

**文件**: `backend/services/reversal_detector.py`

**修改位置**: 第392-430行

**核心逻辑**:
```python
def get_reversal_history(self, limit: int = 100) -> Dict:
    """获取反转历史记录（完全匹配原系统格式）"""
    records = self.db.query(ReversalCondition).order_by(
        ReversalCondition.fetched_at.desc()
    ).limit(limit).all()
    
    items = [{
        'id': r.id,
        'fetched_at': r.fetched_at.isoformat(),
        'gold_price_usd_per_oz': r.gold_price_usd_per_oz,
        'usdcny_rate': r.usdcny_rate,  # 新增字段
        'price_signal': r.price_signal,
        'political_signal': r.political_signal,
        'war_signal': r.war_signal,
        'signal_level': r.signal_level,
        'triggered_conditions': r.triggered_conditions or '',
        'note': r.note or '',  # 新增字段
        'us10y_signal': r.us10y_signal
    } for r in records]
    
    # 计算时间范围（完全匹配原系统）
    range_str = ''
    since_str = ''
    if items:
        latest = items[0]['fetched_at']
        oldest = items[-1]['fetched_at']
        range_str = f"{oldest} ~ {latest}"
        since_str = oldest
    
    return {
        'range': range_str,
        'since': since_str,
        'items': items
    }
```
